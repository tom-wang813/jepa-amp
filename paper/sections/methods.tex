\section{Methods}

\subsection{JEPA Pre-Training}
\label{sec:pretrain}

\paragraph{Architecture.}
The encoder $f_\theta$ is a Transformer~\citep{vaswani} with $d=256$, 4 heads,
4 layers, and $d_\text{ff}=1024$, operating on amino acid sequences tokenised with
a 25-token vocabulary (20 canonical AAs + BOS/EOS/PAD/MASK + unknown).
A separate \emph{target encoder} $\bar{f}_\theta$ (exponential moving average,
EMA momentum~$= 0.996$) produces stable target representations without collapse.

\paragraph{Objective.}
Given a sequence $\mathbf{x}$, we partition tokens into a \emph{context} set
$\mathcal{C}$ (observed prefix) and a \emph{target} set $\mathcal{T}$
(held-out suffix, randomly 30--70\% of tokens).
A lightweight predictor $g_\phi$ maps context representations to predicted
target representations, minimising:
\begin{equation}
  \mathcal{L}_\text{JEPA} =
  \mathbb{E}\bigl[\|g_\phi(f_\theta(\mathbf{x}_\mathcal{C})) -
  \bar{f}_\theta(\mathbf{x}_\mathcal{T})\|_2^2\bigr].
  \label{eq:jepa}
\end{equation}
Unlike masked language models, JEPA predicts in \emph{representation space},
avoiding per-token reconstruction and encouraging learning of higher-level
structural features.

\paragraph{Training.}
We pre-train on \textbf{868{,}724} unique AMP sequences ($\leq 50$ residues) from APD3,
DRAMP, UniProt AMP annotations, and experimental databases, with AMPlify test sequences
excluded to prevent data leakage.
Training: AdamW ($\eta=10^{-3}$, cosine annealing), batch size 2048,
100 epochs, converging to val loss~$=0.244$.

\subsection{Downstream Fine-Tuning}
\label{sec:finetune}

\paragraph{Shared bottleneck adapter.}
All downstream tasks attach a \emph{bottleneck adapter}
$A_\psi: \mathbb{R}^d \to \mathbb{R}^d$ (linear $\to$ GELU $\to$ linear,
bottleneck 128, zero-initialised last layer) on top of the frozen encoder,
adding $<$5\% extra parameters.

\paragraph{AMP classification.}
A two-layer MLP with residual connection maps mean-pooled adapter output to
a binary AMP logit.
The encoder is fine-tuned with differential learning rates
($\eta_\text{head} = 1.5\times10^{-4}$, $\eta_\text{enc} = 3\times10^{-6}$)
to prevent catastrophic forgetting.
Training data mirrors AMPlify (3{,}338 positive + 3{,}338 negative sequences)
for an apple-to-apple comparison.

\paragraph{MIC regression.}
Minimum inhibitory concentrations span 20 bacterial species from GRAMPA~\citep{grampa}.
A Transformer-based head with FiLM species conditioning modulates encoder features
by species identity:
\begin{equation}
  \hat{y}_\text{MIC} = \text{TransHead}
  \bigl((1{+}\gamma_b)\odot A_\psi(f_\theta(\mathbf{x}))+\beta_b\bigr),
\end{equation}
where $(\gamma_b,\beta_b) = W_\text{FiLM}\,e_b$ and $e_b\in\mathbb{R}^{64}$
is a learned bacterial embedding.
Predictions are in $\log_2(\text{MIC}/[\mu\text{g\,mL}^{-1}])$ space.

\paragraph{MC-Dropout uncertainty.}
At inference the encoder is in eval mode but the head remains in train mode.
Running $N=30$ stochastic forward passes through dropout layers yields a mean
estimate and an uncertainty (standard deviation) that correlates with prediction
error ($r=0.44$, $p<10^{-6}$), enabling reliable confidence estimation.

\subsection{Conditional Generation (v4)}
\label{sec:gen}

\paragraph{Condition encoding.}
Sequences are conditioned on a 3-vector
$\mathbf{c}=[\ell/50,\;\tanh(\text{charge}/5),\;\tanh(\text{GRAVY})]$
encoding normalised length, net charge, and Kyte--Doolittle hydrophobicity.
The condition passes through a shared condition MLP to produce a 128-dim embedding
$\mathbf{c}_\text{emb}$, then injected via two pathways (Figure~\ref{fig:arch}C):
\begin{enumerate}
  \item \textbf{Cross-attention token:}
        $\mathbf{c}_\text{tok} = W_\text{proj}\,\mathbf{c}_\text{emb}$
        is prepended to the cross-attention memory.
  \item \textbf{AdaLN per decoder layer:}
        $(\gamma_i,\beta_i) = \text{Linear}_{2d}(\mathbf{c}_\text{emb})$,
        applied as $\text{AdaLN}(h) = (1{+}\gamma_i)\,\text{LN}(h)+\beta_i$.
\end{enumerate}
All AdaLN weights are zero-initialised so training starts from an unconditioned model.

\paragraph{Training losses.}
\begin{equation}
  \mathcal{L} = \underbrace{\mathcal{L}_\text{CE}}_{\text{token prediction}}
  + \alpha\,\underbrace{\mathcal{L}_\text{phys}}_{\text{physicochemical}},
  \quad \alpha = 0.3.
  \label{eq:gen_loss}
\end{equation}
$\mathcal{L}_\text{phys}$ computes expected charge and GRAVY from the predicted
token distribution and penalises deviation from the target:
\begin{equation}
  \mathcal{L}_\text{phys} =
  \operatorname{MSE}\!\Bigl(\sum_t\!\sum_a p_{t,a}\,q_a,\;q^\star\Bigr)
  + \operatorname{MSE}\!\Bigl(\tfrac{1}{T}\sum_t\!\sum_a p_{t,a}\,k_a,\;h^\star\Bigr),
\end{equation}
where $p_{t,a}=\text{softmax}(\text{logits})_{t,a}$, $q_a\in\{-1,0,+1\}$ is the
charge of amino acid $a$, and $k_a$ is its Kyte--Doolittle hydrophobicity.

\paragraph{Context and condition dropout.}
The encoder output is zeroed with $p=0.15$ (\emph{context dropout}) during training
to prevent the decoder from ignoring conditions.
Condition vectors are independently zeroed with $p=0.30$ (\emph{classifier-free guidance
dropout}), enabling guidance-scale controlled inference.

\subsection{MIC-Conditioned Generation (v5)}
\label{sec:mic_gen}

\paragraph{Extended condition vector.}
The condition vector is extended from 3 to \textbf{43 dimensions}:
\begin{equation}
  \mathbf{c} =
  \underbrace{[\ell/50,\;\tanh(q/5),\;\tanh(h)]}_{\text{physchem }(3)}
  \;\|\;
  \underbrace{[m_1,\ldots,m_{20}]}_{\text{MIC values }(20)}
  \;\|\;
  \underbrace{[\mu_1,\ldots,\mu_{20}]}_{\text{MIC mask }(20)},
\end{equation}
where $m_i\in\mathbb{R}$ is the target $\log_2\text{MIC}$ for bacterium $i$
and $\mu_i\in\{0,1\}$ indicates whether species $i$ is constrained.

\paragraph{Training on real MIC labels.}
We fine-tune v5 directly on the GRAMPA database~\citep{grampa},
which provides 51{,}345 experimentally measured (sequence, bacterium, MIC) triples
spanning 3{,}940 unique peptides and 20 species.
Per-sequence MIC vectors are built by averaging replicate measurements;
on average 3.5 of 20 bacteria are observed per sequence, yielding genuinely sparse
supervision.
To prevent overfitting, each sequence is replicated $10\times$ with randomly drawn
prefix splits (35{,}460 effective training examples), and 10\% of revealed MIC values
are randomly withheld per step.
Training converged in 105 epochs (early stopping) with best val loss~$=1.961$.
