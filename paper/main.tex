\section{Architecture Overview}
\label{sec:arch_fig}

\begin{figure}[h]
\centering
\input{figures/architecture.tex}
\caption{JEPA-AMP framework.
\textbf{(A)}~Pre-training: context encoder $f_\theta$ processes observed tokens (blue);
target encoder $\bar{f}_\theta$ (EMA) encodes masked tokens (orange);
predictor $g_\phi$ minimises $\ell_2$ loss in representation space---not token space.
\textbf{(B)}~Fine-tuning: a shared bottleneck adapter (green) is attached to the
frozen encoder. Three task-specific heads share this backbone:
AMP classifier (top), MIC predictor with FiLM species conditioning (middle),
and conditional generator with AdaLN decoder (bottom).
\textbf{(C)}~Conditional generator detail: condition vector $\mathbf{c}$ is processed
by a shared condition MLP and injected both as a cross-attention memory token and via
per-layer AdaLN scale/shift, providing dual gradient pathways that prevent posterior collapse.}
\label{fig:arch}
\end{figure}
